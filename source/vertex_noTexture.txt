#version 300 es

uniform mat4 camera; // Contains projection and view transformations
uniform mat4 model;

in vec3 vert;
in vec3 vertNormal;

out vec3 fragVert;
out vec3 fragNormal;

void main() {
    // Pass the texture coord straight through to the fragment shader
	fragVert = vert;
    fragNormal = vertNormal;

    gl_Position =  camera * model * vec4(vert, 1);
}
